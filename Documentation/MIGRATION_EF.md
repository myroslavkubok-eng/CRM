## 1. Setup & Prerequisites

### Requirements

- [.NET 9 SDK](https://dotnet.microsoft.com/download)
- [PostgreSQL 16+](https://www.postgresql.org/download/) installed locally or on a server
- [EF Core CLI tools](https://learn.microsoft.com/en-us/ef/core/cli/dotnet)

```bash
# Install EF Core CLI tools globally
dotnet tool install --global dotnet-ef

# Verify installation
dotnet ef --version
```

### Create the Database

```sql
-- Connect to PostgreSQL and create the database
CREATE DATABASE katia;
CREATE USER katia_user WITH ENCRYPTED PASSWORD 'your_secure_password';
GRANT ALL PRIVILEGES ON DATABASE katia TO katia_user;
```

**`Katia.Infrastructure/Data/ApplicationDbContext.cs`**
```csharp
using Microsoft.AspNetCore.Identity;
using Microsoft.AspNetCore.Identity.EntityFrameworkCore;
using Microsoft.EntityFrameworkCore;

public class ApplicationDbContext : IdentityDbContext<User, IdentityRole<Guid>, Guid>
{
    public ApplicationDbContext(DbContextOptions<ApplicationDbContext> options)
        : base(options) { }

    public DbSet<Salon> Salons => Set<Salon>();
    public DbSet<Service> Services => Set<Service>();
    public DbSet<Master> Masters => Set<Master>();
    public DbSet<MasterService> MasterServices => Set<MasterService>();
    public DbSet<Client> Clients => Set<Client>();
    public DbSet<Booking> Bookings => Set<Booking>();
    public DbSet<BookingStatusHistory> BookingStatusHistories => Set<BookingStatusHistory>();
    public DbSet<BookingPayment> BookingPayments => Set<BookingPayment>();
    public DbSet<DepositSettings> DepositSettings => Set<DepositSettings>();
    public DbSet<GiftCard> GiftCards => Set<GiftCard>();
    public DbSet<GiftCardUsage> GiftCardUsages => Set<GiftCardUsage>();
    public DbSet<Notification> Notifications => Set<Notification>();
    public DbSet<FeedPost> FeedPosts => Set<FeedPost>();
    public DbSet<SalonInvitation> SalonInvitations => Set<SalonInvitation>();

    protected override void OnModelCreating(ModelBuilder builder)
    {
        base.OnModelCreating(builder);
        builder.ApplyConfigurationsFromAssembly(typeof(ApplicationDbContext).Assembly);
    }

    // Auto-update UpdatedAt on save
    public override Task<int> SaveChangesAsync(CancellationToken cancellationToken = default)
    {
        var updatedEntries = ChangeTracker.Entries()
            .Where(e => e.State == EntityState.Modified);

        foreach (var entry in updatedEntries)
        {
            if (entry.Entity is Salon salon) salon.UpdatedAt = DateTime.UtcNow;
            if (entry.Entity is GiftCard card) card.UpdatedAt = DateTime.UtcNow;
            if (entry.Entity is Booking booking) booking.UpdatedAt = DateTime.UtcNow;
        }

        return base.SaveChangesAsync(cancellationToken);
    }
}
```

```csharp
public class SalonConfiguration : IEntityTypeConfiguration<Salon>
{
    public void Configure(EntityTypeBuilder<Salon> builder)
    {
        builder.ToTable("salons");
        builder.HasKey(s => s.Id);

        builder.Property(s => s.Name).IsRequired().HasMaxLength(200);
        builder.Property(s => s.Email).HasMaxLength(200);
        builder.Property(s => s.Phone).HasMaxLength(50);

        // Store Photos as JSON array in PostgreSQL
        builder.Property(s => s.Photos)
            .HasColumnType("jsonb");

        // Enum stored as string
        builder.Property(s => s.Plan)
            .HasConversion<string>();
        builder.Property(s => s.BillingPeriod)
            .HasConversion<string>();
        builder.Property(s => s.SubscriptionStatus)
            .HasConversion<string>();

        // Relationships
        builder.HasOne(s => s.Owner)
            .WithMany(u => u.OwnedSalons)
            .HasForeignKey(s => s.OwnerId)
            .OnDelete(DeleteBehavior.Restrict);

        builder.HasMany(s => s.Services)
            .WithOne(svc => svc.Salon)
            .HasForeignKey(svc => svc.SalonId)
            .OnDelete(DeleteBehavior.Cascade);

        builder.HasMany(s => s.Masters)
            .WithOne(m => m.Salon)
            .HasForeignKey(m => m.SalonId)
            .OnDelete(DeleteBehavior.Cascade);

        builder.HasOne(s => s.DepositSettings)
            .WithOne(d => d.Salon)
            .HasForeignKey<DepositSettings>(d => d.SalonId)
            .OnDelete(DeleteBehavior.Cascade);

        // Indexes
        builder.HasIndex(s => s.OwnerId);
        builder.HasIndex(s => s.IsPublished);
    }
}
```

```csharp
        builder.HasIndex(b => b.StartTime);
```

## 7. Migrations

### Register the DbContext in Program.cs

```csharp
builder.Services.AddDbContext<ApplicationDbContext>(options =>
    options.UseNpgsql(
        builder.Configuration.GetConnectionString("DefaultConnection"),
        npgsql => npgsql.MigrationsAssembly("Katia.Infrastructure")
    )
    .UseSnakeCaseNamingConvention()  // converts PascalCase → snake_case for PostgreSQL
);
```

> Install `EFCore.NamingConventions` to enable `UseSnakeCaseNamingConvention()`:
> ```bash
> dotnet add package EFCore.NamingConventions
> ```

### Run Migrations

```bash
# Create the initial migration (run from solution root)
dotnet ef migrations add InitialCreate \
  --project src/Katia.Infrastructure \
  --startup-project src/Katia.API \
  --output-dir Data/Migrations

# Apply migration to the database
dotnet ef database update \
  --project src/Katia.Infrastructure \
  --startup-project src/Katia.API

# Future migrations (example: adding a column)
dotnet ef migrations add AddSalonRating \
  --project src/Katia.Infrastructure \
  --startup-project src/Katia.API

dotnet ef database update \
  --project src/Katia.Infrastructure \
  --startup-project src/Katia.API
```

### Useful Migration Commands

```bash
# See pending migrations
dotnet ef migrations list --project src/Katia.Infrastructure --startup-project src/Katia.API

# Rollback last migration
dotnet ef migrations remove --project src/Katia.Infrastructure --startup-project src/Katia.API

# Rollback to a specific migration
dotnet ef database update PreviousMigrationName --project src/Katia.Infrastructure --startup-project src/Katia.API

# Generate SQL script (useful for production deployments)
dotnet ef migrations script --project src/Katia.Infrastructure --startup-project src/Katia.API -o migration.sql
```

## 11. Repositories Pattern

Use repositories to keep database logic out of controllers.

**`Katia.Infrastructure/Repositories/BookingRepository.cs`**
```csharp
public class BookingRepository : IBookingRepository
{
    private readonly ApplicationDbContext _db;

    public BookingRepository(ApplicationDbContext db) => _db = db;

    public async Task<Booking?> GetByIdAsync(Guid id) =>
        await _db.Bookings
            .Include(b => b.Master)
            .Include(b => b.Service)
            .Include(b => b.Payment)
            .Include(b => b.StatusHistory.OrderByDescending(h => h.ChangedAt))
            .FirstOrDefaultAsync(b => b.Id == id);

    public async Task<List<Booking>> GetBySalonAsync(Guid salonId, BookingStatus? status = null)
    {
        var query = _db.Bookings
            .Where(b => b.SalonId == salonId)
            .Include(b => b.Master)
            .Include(b => b.Service)
            .AsQueryable();

        if (status.HasValue)
            query = query.Where(b => b.Status == status.Value);

        return await query
            .OrderByDescending(b => b.StartTime)
            .ToListAsync();
    }

    public async Task<bool> HasConflictAsync(Guid masterId, DateTime start, DateTime end, Guid? excludeBookingId = null)
    {
        var query = _db.Bookings
            .Where(b => b.MasterId == masterId
                     && b.Status != BookingStatus.Cancelled
                     && b.Status != BookingStatus.DeclinedBySalon
                     && b.Status != BookingStatus.Expired
                     && b.StartTime < end
                     && b.EndTime > start);

        if (excludeBookingId.HasValue)
            query = query.Where(b => b.Id != excludeBookingId.Value);

        return await query.AnyAsync();
    }

    public async Task<List<Booking>> GetExpiredPendingAsync() =>
        await _db.Bookings
            .Where(b => b.Status == BookingStatus.Pending
                     && b.ConfirmationDeadline < DateTime.UtcNow)
            .ToListAsync();

    public async Task AddAsync(Booking booking)
    {
        _db.Bookings.Add(booking);
        await _db.SaveChangesAsync();
    }

    public async Task SaveChangesAsync() =>
        await _db.SaveChangesAsync();
}
```

**Register in `Program.cs`:**
```csharp
builder.Services.AddScoped<IBookingRepository, BookingRepository>();
builder.Services.AddScoped<ISalonRepository, SalonRepository>();
builder.Services.AddScoped<IGiftCardRepository, GiftCardRepository>();
```

## 12. Query Examples

### Get salon with masters and services
```csharp
var salon = await _db.Salons
    .Include(s => s.Masters)
        .ThenInclude(m => m.MasterServices)
            .ThenInclude(ms => ms.Service)
    .Include(s => s.Services)
    .Include(s => s.DepositSettings)
    .FirstOrDefaultAsync(s => s.Id == salonId);
```

### Get bookings for a date range
```csharp
var bookings = await _db.Bookings
    .Where(b => b.SalonId == salonId
             && b.StartTime >= startDate
             && b.StartTime < endDate
             && b.Status != BookingStatus.Cancelled)
    .Include(b => b.Master)
    .Include(b => b.Service)
    .OrderBy(b => b.StartTime)
    .ToListAsync();
```

### Gift card analytics
```csharp
var analytics = await _db.GiftCards
    .Where(g => g.SalonId == salonId)
    .GroupBy(g => g.Status)
    .Select(g => new { Status = g.Key, Count = g.Count(), Revenue = g.Sum(x => x.Amount) })
    .ToListAsync();
```

### Client spending history
```csharp
var clientStats = await _db.Clients
    .Where(c => c.SalonId == salonId)
    .Select(c => new {
        c.Id,
        c.FirstName,
        c.LastName,
        TotalBookings = c.Bookings.Count(b => b.Status == BookingStatus.Completed),
        TotalSpent = c.Bookings
            .Where(b => b.Status == BookingStatus.Completed && b.Payment != null)
            .Sum(b => b.Payment!.TotalAmount)
    })
    .OrderByDescending(c => c.TotalSpent)
    .ToListAsync();
```

### Auto-expire pending bookings (Hangfire job)
```csharp
var expired = await _db.Bookings
    .Where(b => b.Status == BookingStatus.Pending
             && b.ConfirmationDeadline < DateTime.UtcNow)
    .ToListAsync();

foreach (var booking in expired)
    booking.Status = BookingStatus.Expired;

await _db.SaveChangesAsync();
```
